### C. Deadlift

### D. Shoulder Press

### Angle Calculation
Use `math.atan2` for robust angle calculation between three points (A, B, C) where B is the joint.
```python
import math

def calculate_angle(a, b, c):
    """Calculates angle at point b given points a, b, c in radians then degrees."""
    radians = math.atan2(c[1]-b[1], c[0]-b[0]) - math.atan2(a[1]-b[1], a[0]-b[0])
    angle = abs(radians * 180.0 / math.pi)
    if angle > 180.0:
        angle = 360 - angle
    return angle
```

### 5-Frame Anti-Jitter State Machine
Do not increment the rep counter immediately. Use a state machine.
```python
class RepCounter:
    def __init__(self, threshold_low, threshold_high):
        self.threshold_low = threshold_low
        self.threshold_high = threshold_high
        self.state = "UP" # or "DOWN"
        self.consecutive_frames = 0
        self.reps = 0

    def update(self, current_angle):
        self.consecutive_frames += 1
        
        if self.state == "UP" and current_angle < self.threshold_low:
            # Entered "DOWN" state
            self.state = "DOWN"
            self.consecutive_frames = 0 # Reset counter for new state
        elif self.state == "DOWN" and current_angle > self.threshold_high:
            # Entered "UP" state -> Complete Rep
            if self.consecutive_frames >= 5:
                self.reps += 1
                self.state = "UP"
                self.consecutive_frames = 0
        
        return self.reps, self.state
```

### Smoothing
Apply a **Moving Average** (window size = 5) to raw landmark coordinates before calculating angles to reduce noise from the webcam feed.
```python
from collections import deque

class LandmarkSmoother:
    def __init__(self, window_size=5):
        self.window_size = window_size
        self.x_history = deque(maxlen=window_size)
        self.y_history = deque(maxlen=window_size)
    
    def smooth(self, x, y):
        self.x_history.append(x)
        self.y_history.append(y)
        smooth_x = sum(self.x_history) / len(self.x_history)
        smooth_y = sum(self.y_history) / len(self.y_history)
        return smooth_x, smooth_y
```

```python
class PoseTracker:
    def __init__(self):
        self.mp_pose = mp.solutions.pose
        self.pose = self.mp_pose.Pose()
        self.mp_drawing = mp.solutions.drawing_utils
        self.exercise = "bicep_curl"
        self.left_counter = RepCounter(65, 165)
        self.right_counter = RepCounter(65, 165)
        self.left_smoother = LandmarkSmoother(5)
        self.right_smoother = LandmarkSmoother(5)
        self.form_score = 100
        self.errors = []
        self.frame_count = 0
    
    def detect_landmarks(self, frame):
        """Detect pose landmarks from frame."""
        rgb_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)
        results = self.pose.process(rgb_frame)
        return results
    
    def get_landmark_coords(self, results, landmark_id):
        """Extract x, y coordinates from landmark."""
        if results.pose_landmarks:
            landmark = results.pose_landmarks.landmark[landmark_id]
            return landmark.x, landmark.y
        return None, None
    
    def process_frame(self, frame):
        """Process a single frame and return rep data."""
        results = self.detect_landmarks(frame)
        
        if not results.pose_landmarks:
            return {
                "status": "camera_error",
                "errors": ["landmarks_not_detected"]
            }
        
        # Get landmarks for current exercise
        if self.exercise == "bicep_curl":
            return self.process_bicep_curl(results)
        elif self.exercise == "squat":
            return self.process_squat(results)
        elif self.exercise == "deadlift":
            return self.process_deadlift(results)
        elif self.exercise == "shoulder_press":
            return self.process_shoulder_press(results)
        elif self.exercise == "lat_pulldown":
            return self.process_lat_pulldown(results)
        elif self.exercise == "plank":
            return self.process_plank(results)
    
    def process_bicep_curl(self, results):

        # Left arm
        left_shoulder = self.get_landmark_coords(results, 11)
        left_elbow = self.get_landmark_coords(results, 13)
        left_wrist = self.get_landmark_coords(results, 15)
        
        # Right arm
        right_shoulder = self.get_landmark_coords(results, 12)
        right_elbow = self.get_landmark_coords(results, 14)
        right_wrist = self.get_landmark_coords(results, 16)
        
        # Smooth coordinates

        right_elbow_smooth = self.right_smoother.smooth(right_elbow[0], right_elbow[1])

        left_angle = calculate_angle(left_shoulder, left_elbow_smooth, left_wrist)
        right_angle = calculate_angle(right_shoulder, right_elbow_smooth, right_wrist)
        
        # Update counters
        left_reps, left_stage = self.left_counter.update(left_angle)
        right_reps, right_stage = self.right_counter.update(right_angle)
        
        total_reps = left_reps + right_reps
        
        return {
            "exercise": "bicep_curl",
            "total_reps": total_reps,
            "left_reps": left_reps,
            "right_reps": right_reps,
            "stage": left_stage,
            "current_angle": (left_angle + right_angle) / 2,
            "form_score": 90,
            "errors": [],
            "status": "active"
```

### WebSocket Server
```python
async def handle_client(websocket, path):
    """Handle WebSocket connections."""
    try:
        async for message in websocket:
            data = json.loads(message)
            if data.get("command") == "set_exercise":
                tracker.exercise = data.get("exercise", "bicep_curl")
            elif data.get("command") == "start":
                await stream_frames(websocket)
    except Exception as e:
        print(f"Error: {e}")

async def stream_frames(websocket):
    """Stream pose data to frontend."""
    cap = cv2.VideoCapture(0)
    while True:
        ret, frame = cap.read()
        if not ret:
            await websocket.send(json.dumps({"status": "camera_error"}))
            break
        
        data = tracker.process_frame(frame)
        await websocket.send(json.dumps(data))
        await asyncio.sleep(0.03)  # ~30 FPS
    cap.release()

async def main():
    async with serve(handle_client, "0.0.0.0", 8765):
        print("WebSocket server running on ws://0.0.0.0:8765")
        await asyncio.Future()

if __name__ == "__main__":
    tracker = PoseTracker()
    asyncio.run(main())
```
